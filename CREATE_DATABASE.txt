create or replace FUNCTION isPojazdAlreadyUsed (v_vin IN VARCHAR2)
return integer is

v_flag integer := 0;

begin

SELECT
   CASE WHEN EXISTS(SELECT 1 FROM Pojazd where nr_VIN = v_vin)
     THEN 1
     ELSE 0
   END
into v_flag FROM dual;

return v_flag;
end;
