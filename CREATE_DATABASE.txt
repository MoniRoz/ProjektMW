CREATE TABLE Posiadanie (ID_posiadania number(10) NOT NULL, ID_pojazdu number(10) NOT NULL, ID_wlasciciela number(10) NOT NULL, data_poczatkowa date NOT NULL, data_koncowa date NOT NULL, PRIMARY KEY (ID_posiadania));

CREATE TABLE Marka (ID_marki number(10) NOT NULL, marka varchar2(50) NOT NULL, PRIMARY KEY (ID_marki));
CREATE TABLE Rodzaj (ID_rodzaju number(10) NOT NULL, rodzaj varchar2(50) NOT NULL, PRIMARY KEY (ID_rodzaju));
CREATE TABLE Przeglad (ID_przegladu number(10) NOT NULL, ID_dokument number(10) NOT NULL, data_wystawienia date NOT NULL, data_waznosci date NOT NULL, wystawiajacy varchar2(50) NOT NULL, PRIMARY KEY (ID_przegladu));
